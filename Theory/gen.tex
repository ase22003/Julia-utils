\subsection{Pattern Matching}

A \textit{pattern} is a map \(\mathfrak{P} : D \to \{\top, \bot\}\), for any set \(D\). A \textit{classification} is a map \(C : D \to \mathcal{P}(P)\), where \(P\) is a set of patterns and
%\(C(d) = M\) where \(d\in D\) 
%%, \(M\in\mathcal{P}(P)\) 
%and \(\forall p\in M\,\left(p(d)\right)\).
\(C(d) = M \implies \forall p\in M\,\left(p(d)\right)\), where \(d\in D\).

No.

A \textit{pattern} is a set \(p\). If a set, vector, element (or other data) \(d\) maches the pattern \(p\), then \(\Omega \in p\). The set \(\left\{p\,\middle|\,\Omega\in p\right\}\) is the set of patterns to which some data \(\Omega\) belongs.
\[
\begin{array}{ccl}
	\mathfrak{E} & := & \setdef{\Omega\in D}{\Omega\in\mathfrak{T} \lor \left(\Omega=\vect{\omega_1, \cdots, \omega_n, \text{``\(+\)''}, \omega_k, \cdots, \omega_j} \land \set{\vect{\omega_1, \cdots, \omega_n}, \vect{\omega_k, \cdots, \omega_j}} \subseteq \mathfrak{E}\right)} \\
	& = & \setdef{\Omega\in D}{\Omega\in\mathfrak{T} \lor \left(\Omega\in\setdef{\vect{\omega_1, \cdots, \omega_n, \text{``\(+\)''}, \omega_k, \cdots, \omega_j}\in D}{\set{\vect{\omega_1, \cdots, \omega_n}, \vect{\omega_k, \cdots, \omega_j}} \subseteq \mathfrak{E}}\right)} \\
	& = & \setdef{\Omega\in D}{\Omega\in\mathfrak{T} \cup \setdef{\vect{\omega_1, \cdots, \omega_n, \text{``\(+\)''}, \omega_k, \cdots, \omega_j}}{\set{\vect{\omega_1, \cdots, \omega_n}, \vect{\omega_k, \cdots, \omega_j}} \subseteq \mathfrak{E}}} \\
	& = & \mathfrak{T} \cup \setdef{\vect{\omega_1, \cdots, \omega_n, \text{``\(+\)''}, \omega_k, \cdots, \omega_j}\in D}{\set{\vect{\omega_1, \cdots, \omega_n}, \vect{\omega_k, \cdots, \omega_j}} \subseteq \mathfrak{E}} \\
	\mathfrak{T} & := & \setdef{\Omega\in D}{\Omega\in\mathfrak{F} \lor \left(\Omega=\vect{\omega_1, \cdots, \omega_n, \text{``\(\times\)''}, \omega_k, \cdots, \omega_j} \land \set{\vect{\omega_1, \cdots, \omega_n}, \vect{\omega_k, \cdots, \omega_j}} \subseteq \mathfrak{T}\right)} \\
	\mathfrak{F} & := & \setdef{\Omega\in D}{\Omega\in\mathfrak{N} \lor \left(\Omega=\vect{\text{``(''}, \omega_1, \cdots, \omega_n, \text{``)''}} \land \vect{\omega_1, \cdots, \omega_n} \in \mathfrak{E}\right)} \\
	\mathfrak{N} & := & \setdef{\Omega\in D}{\Omega \in \set{\text{``0''}, \cdots, \text{``9''}} \lor \left( \Omega = \vect{\omega_1, \omega_2, \cdots, \omega_n} \wedge \omega_1\in\mathfrak{N} \wedge \vect{\omega_2, \cdots, \omega_n}\in\mathfrak{N}\right)} \\
	%\mathfrak{N} & := & \setdef{\Omega\in D}{\Omega \in \set{\text{``0''}, \cdots, \text{``9''}} \lor \Omega\in\mathfrak{N}}
\end{array}
\]

\[
\begin{array}{cl}
	\mathfrak{E}(\Omega) = &\left\{
		\begin{array}{cl}
			\top & : {\begin{array}{cl}
				     & \Omega = \vect{\omega_1, \cdots, \omega_n, \text{``\(+\)''}, \omega_k, \cdots, \omega_j} \wedge \mathfrak{E}\left(\vect{\omega_1, \cdots, \omega_n}\right) \wedge \mathfrak{E}\left(\vect{\omega_k, \cdots, \omega_j}\right) \\
				\vee & \mathfrak{T}(\Omega)
			\end{array}} \\
			\bot & : \text{otherwise}
		\end{array}
	\right. \\
	\mathfrak{T}(\Omega) = &\left\{
		\begin{array}{cl}
			\top & : {\begin{array}{cl}
				     & \Omega = \vect{\omega_1, \cdots, \omega_n, \text{``\(\times\)''}, \omega_k, \cdots, \omega_j} \wedge \mathfrak{T}\left(\vect{\omega_1, \cdots, \omega_n}\right) \wedge \mathfrak{T}\left(\vect{\omega_k, \cdots, \omega_j}\right) \\
				\vee & \mathfrak{F}(\Omega)
			\end{array}} \\
			\bot & : \text{otherwise}
		\end{array}
	\right. \\
	\mathfrak{F}(\Omega) = &\left\{
		\begin{array}{cl}
			\top & : {\begin{array}{cl}
				     & \Omega = \vect{\text{``(''}, \omega_1, \cdots, \omega_n, \text{``)''}} \wedge  \mathfrak{E}\left(\vect{\omega_1, \cdots, \omega_n}\right) \\
				\vee & \mathfrak{N}\vect{\Omega}
			\end{array}} \\
			\bot & : \text{otherwise}
		\end{array}
	\right. \\
	\mathfrak{N}(\Omega) = &\left\{
		\begin{array}{cl}
			\top & : {\begin{array}{cl}
				     & \Omega \in \set{\text{``0''}, \text{``1''}, \text{``2''}, \cdots, \text{``9''}} \\

			\end{array}} \\
			\bot & : \text{otherwise}
		\end{array}
	\right. \\
\end{array}
